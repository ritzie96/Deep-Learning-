        self.size = images.shape[1]
        self.images = images
        self.one_hot_label = one_hot_label
        self.num_label = num_label
        self.set_name = set_name

        return Bat(
            images=self.images[:, :new_size],

            set_name=self.set_name
        )

    def mean(self):
        return self.images.mean(axis=1)

class dataset:
    size_o = 10
    size_i = 32 * 32 * 3

    def __init__(self):
        self.labels = self.load_labels()
        self.loadbatch = {}
        self.encode_label = preprocessing.LabelBinarizer()
        self.encode_label.fit([x for x in range(self.size_o)])
    @staticmethod
    def load_labels():
        with open('cifar-10-batches-py/batches.meta', 'rb') as f:
            data = pickle.load(f, encoding='bytes')
            return [x.decode('ascii') for x in data[b'label_names']]

    def getbat(self, batch_name) -> Bat:
        if batch_name not in self.loadbatch:
            with open('cifar-10-batches-py/' + batch_name, 'rb') as f:
                data = pickle.load(f, encoding='bytes')

            self.loadbatch[batch_name] = Bat(
                images=np.divide(data[b'data'], 255).T,
                one_hot_label=self.encode_label.transform(data[b'labels']).T,
                num_label=data[b'labels'],
                set_name=self.labels
            )
        return self.loadbatch[batch_name]

    def get_batch(self, *args) -> Bat:
        batches = [self.getbat(name) for name in args]
        return Bat(
            images=np.hstack([b.images for b in batches]),
            one_hot_label=np.hstack([b.one_hot_label for b in batches]),
            num_label=np.hstack([b.num_label for b in batches]),
            set_name=self.labels
        )

    def disp_img(self, img, label: int = None, interpolation='gaussian'):
        sqimg = np.rot90(np.reshape(img, (32, 32, 3), order='F'), k=3)
        plt.imshow(sqimg, interpolation=interpolation)
        plt.axis('off')
        plt.title(self.labels[label] if label is not None else '')

class Initial(ABC):
    @abstractmethod
    def new_matrix(self, shape):
        pass

class Zeros(Initial):
    def new_matrix(self, shape):
        return np.zeros(shape, dtype=float)

class XavierInit(Initial):
    def new_matrix(self, shape):
        var = 2 / (shape[0] + shape[1])
        return np.random.normal(0, np.sqrt(var), shape)

        self.size_i = size_i
        self.size_o = size_o

    def __str__(self):

class Linear(Layer):
    def __init__(self, size_i, size_o, w_lamda,
                 w_init: Initial = None,
                 b_init: Initial = None,
                 name='Linear'):

        super().__init__(size_i, size_o, name)
        self.w_lamda = w_lamda
        w_init = w_init or Zeros()
        b_init = b_init or Zeros()
        self.Weight = w_init.new_matrix((self.size_o, self.size_i))
        self.bias = b_init.new_matrix((self.size_o, 1))
        self.grad_W = np.empty_like(self.Weight, dtype=float)
        self.grad_b = np.empty_like(self.bias, dtype=float)
        self.X = np.empty(shape=(self.size_i, 1))

    def fwd(self, X):
        self.X = X
        return np.dot(self.Weight, X) + self.bias

    def bwd(self, grad):
        assert self.X.shape[1] == grad.shape[0]
        N = self.X.shape[1]
        self.grad_W = np.zeros_like(self.Weight, dtype=float)
        self.grad_b = np.zeros_like(self.bias, dtype=float)
        for i in range(N):
            x = self.X[:, i]
            g = grad[i, :]
            self.grad_W += np.outer(g, x)
            self.grad_b += np.reshape(g, self.grad_b.shape)

        self.grad_W = self.grad_W / N + 2 * self.w_lamda * self.Weight
        self.grad_b /= N
        return np.dot(grad, self.Weight)

    def cost(self):
        return self.w_lamda * np.power(self.Weight, 2).sum()

class ReLU(Layer):
    def __init__(self, size_i, name='ReLU'):
        super().__init__(size_i, size_i, name)
        self.X = np.empty(shape=(self.size_i, 1))

    def fwd(self, X):
        self.X = X
        return X * (X > 0)

    def bwd(self, grad):
        assert self.X.shape[1] == grad.shape[0]
        return np.multiply(grad, self.X.T > 0)

class Softmax(Layer):
    def __init__(self, size_i, name='Softmax'):
        super().__init__(size_i, size_i, name)
        self.P = np.empty(shape=(self.size_i, 1))

    def fwd(self, X):
        try:
            e = np.exp(X)
            P = e / np.sum(e, axis=0)
        except FloatingPointError:
            P = self.softmax_cbyc(X)
        self.P = P
        return P

    def bwd(self, one_hot_target):
        N = self.P.shape[1]
        N = one_hot_target.shape[1]
        output_target_pairs = ((one_hot_target[:, i], self.P[:, i]) for i in range(N))
        grad = (self.softmax_gradient(y, p) for (y, p) in output_target_pairs)
        return np.vstack(grad)

    def cost(self, one_hot_target, P=None):
        if P is None:
            P = self.P
        N = P.shape[1]
        N = one_hot_target.shape[1]
        log_arg = np.multiply(one_hot_target, P).sum(axis=0)
        log_arg[log_arg == 0] = np.finfo(float).eps
        return - np.log(log_arg).sum() / N

    def softmax_cbyc(self, X):
        N = X.shape[1]
        res = np.full_like(X, fill_value=np.finfo(float).eps)
        for i in range(N):
            x = X[:, i]
            try:
                e = np.exp(x)
                res[:, i] = e / e.sum()
            except FloatingPointError:
                res[np.argmax(x), :] = 1 - (self.size_i - 1) * np.finfo(float).eps
        return res

    @staticmethod
    def softmax_gradient(y, p):
        t1 = np.outer(p, p)
        t2 = np.dot(y, (np.diag(p) - t1))
        t3 = np.dot(y, p)
        if t3 == 0:
            t3 = np.finfo(float).eps

        return - t2 / t3

class Bat_Norm(Layer):
    def __init__(self, size_i, mu=None, Sigma=None, name='Bat_Norm'):
        super().__init__(size_i, size_i, name)
        self.mu = mu if mu is not None else np.zeros(shape=(size_i, 1), dtype=float)
        self.Sigma = Sigma if Sigma is not None else np.eye(size_i, dtype=float)

    def fwd(self, X, train=False):
        if train:
            mu = X.mean(axis=1)
            Sigma = X.var(axis=1)
        else:
            mu = self.mu
            Sigma = self.Sigma
        return np.dot(Sigma, (X.T - mu.T).T)

    def bwd(self, grad):
        return grad

class Net:
    def __init__(self):
        self.layers = []

    def add_layer(self, layer: Layer):
        self.layers.append(layer)

    def evaluation(self, inputs):
        outputs = inputs
        for layer in self.layers:
            outputs = layer.fwd(outputs)
        return outputs

    def bwd(self, one_hot_target, inputs=None):
        if inputs is not None:
            self.evaluation(inputs)

        for layer in self.layers[::-1]:
            if type(layer) is Softmax:
                grad = layer.bwd(one_hot_target)
            else:
                grad = layer.bwd(grad)

    def cost(self, one_hot_target, inputs=None, outputs=None):
        cost = 0
        if outputs is None:
            outputs = self.evaluation(inputs)
        for layer in self.layers:
            if type(layer) is Softmax:
                cost += layer.cost(one_hot_target, outputs)
            else:
                cost += layer.cost()

        return cost

    def acc(self, one_hot_target, inputs=None, outputs=None):
        N = one_hot_target.shape[1]
        if outputs is None:
            outputs = self.evaluation(inputs)

        predicted_labels = np.argmax(outputs, axis=0)
        true_labels = np.argmax(one_hot_target, axis=0)
        return np.sum(predicted_labels == true_labels) / N

    def cost_accuracy(self, setdata):
        Y = self.evaluation(setdata.images)
        return (self.cost(setdata.one_hot_label, None, Y),
                self.acc(setdata.one_hot_label, None, Y))

def finite_diff_num(one_hot_target, inputs, matrix, network: Net, initial_cost):
    analytic_grad = np.zeros_like(matrix)
    h = 1e-6  
    desc = 'Computing gradients for {} matrix, {} sample' \
        .format(matrix.shape, inputs.shape[1])
    with tqdm(desc=desc, total=matrix.size) as progress_bar:
        for i in range(matrix.shape[0]):
            for j in range(matrix.shape[1]):
                matrix[i, j] += h
                final_cost = network.cost(one_hot_target, inputs)
                analytic_grad[i, j] = (final_cost - initial_cost) / h
                matrix[i, j] -= h
                progress_bar.update()
    return analytic_grad

class vanillasgd(optimize):
    def __init__(self, network, eta_initial,
    eta_decay=1.0, shuffle=False):
        super().__init__(network)
        self.learning_rate = eta_initial
        self.eta_decay = eta_decay
        self.shuffle = shuffle
        self.learning_rates = []

    def train(self, tset, vset, epochs=1, batch_size=None):
        batch_size = batch_size or tset.size

        for epoch_num in tqdm(range(epochs), desc='Epochs'):
            self.train_epoch(tset, batch_size)
            self.update_metrics(tset, vset, epoch_num)
            self.learning_rate *= self.eta_decay

    def train_epoch(self, tset, batch_size):
        if self.shuffle and (batch_size != tset.size):
            indexes = np.random.permutation(tset.size)
        else:
            indexes = np.arange(tset.size)

        for start in range(0, tset.size, batch_size):
            self.train_batch(tset, indexes[start: start + batch_size])
            self.execute_update()

    def train_batch(self, tset, indexes):
        inputs = tset.images[:, indexes]
        one_hot_target = tset.one_hot_label[:, indexes]

        self.network.evaluation(inputs)
        self.network.bwd(one_hot_target)

    def execute_update(self):
        for layer in self.network.layers:
            if type(layer) is Linear:
                layer.Weight -= self.learning_rate * layer.grad_W
                layer.bias -= self.learning_rate * layer.grad_b

    def update_metrics(self, tset, vset, epoch_num):
        super().update_metrics(tset, vset, epoch_num)
        self.learning_rates.append(self.learning_rate)

class momentumsgd(vanillasgd):
    def __init__(self, network, eta_initial, eta_decay=0.95, shuffle=False, rho=0.0):
        super().__init__(network, eta_initial, eta_decay, shuffle)
        self.rho = rho

        self.updatables = [{
                               'layer': layer,
                               'w_rho': np.zeros_like(layer.Weight),
                               'b_rho': np.zeros_like(layer.bias)
                           }
                           for layer in self.network.layers
                           if type(layer) is Linear]

    def execute_update(self):
        for his in self.updatables:
            his['w_rho'] = \
                self.rho * his['w_rho'] + \
                self.learning_rate * his['layer'].grad_W
            his['b_rho'] = \
                self.rho * his['b_rho'] + \
                self.learning_rate * his['layer'].grad_b

            his['layer'].Weight -= his['w_rho']
            his['layer'].bias -= his['b_rho']

def net_train(tset: Bat,
                     vset: Bat,
                     epochs: int,
                     hidden_size: int,
                     lamda: float,
                     eta_initial: float,
                     eta_decay: float,
                     rho: float,
                     train_id: str,
                     test: Bat = None):

    mu = tset.mean()

    net = Net()
    net.add_layer(Bat_Norm(dataset.size_i, mu))
    net.add_layer(Linear(dataset.size_i, hidden_size, lamda, XavierInit()))
    net.add_layer(ReLU(hidden_size))
    net.add_layer(Linear(hidden_size, dataset.size_o, lamda, XavierInit()))
    net.add_layer(Softmax(dataset.size_o))
    opt = momentumsgd(net, eta_initial, eta_decay, True, rho)
    opt.train(tset, vset, epochs, 10000)
    plot = cal_costplot(opt.epoch_nums, opt.t_accuracy, opt.v_accuracy,
                                 opt.t_cost, opt.v_cost,
                                 'plotresults/{}.png'.format(train_id))

    result = {
        'epochs': epochs,
        'hidden_size': hidden_size,
        'lamda': lamda,
        'eta_initial': eta_initial,
        'eta_decay': eta_decay,
        'rho': rho,
        'epoch_nums': opt.epoch_nums,
        't_cost': opt.t_cost,
        't_accuracy': opt.t_accuracy,
        'v_cost': opt.v_cost,
        'v_accuracy': opt.v_accuracy,
        'final_t_cost': opt.t_cost[-1],
        'final_t_accuracy': opt.t_accuracy[-1],
        'final_v_cost': opt.v_cost[-1],
        'final_v_accuracy': opt.v_accuracy[-1],
        'plot': plot
    }

    if test is not None:
        result['final_cost_test'], result['final_acc_test'] = net.cost_accuracy(test)

    return result
