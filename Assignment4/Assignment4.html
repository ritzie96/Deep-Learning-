class dataset:
    def __init__(self, fname):
        self.c_seq = list(self.read_fname(fname))
        self.enc_label = preprocessing.LabelBinarizer()
        self.enc_text = self.enc_label.fit_transform(
            self.c_seq)
        self.char_tot, self.classes_no = self.enc_text.shape

    def enc(self, *val):
        if len(val) == 1:
            return np.squeeze(self.enc_label.transform(list(val[0])))
        else:
            return [self.enc(s) for s in val]

    def dec(self, *seqs):
        if len(seqs) == 1:
            return ''.join(self.enc_label.inverse_transform(seqs[0]))
        else:
            return [self.dec(s) for s in seqs]

    def getseqs(self, length=25):
        for i in range(0, self.char_tot - length, length):
            yield ITSeq(
                ip=self.enc_text[i:i + length],
                op=self.enc_text[i + 1:i + length + 1]
            )

    @staticmethod
    def read_fname(fname):
        with open(fname, 'r') as f:
            return f.read()

class Initial(ABC):
    @abstractmethod
    def new_matrix(self, shape):
        pass

class Zeros(Initial):
    def new_matrix(self, shape):
        return np.zeros(shape, dtype=float)

class XavierInit(Initial):
    def new_matrix(self, shape):
        var = 2 / (sum(shape))
        return np.random.normal(0, np.sqrt(var), shape)

class Rneuralnet:
    def __init__(self, in_size, out_size, st_size,
                 w_init, u_init, v_init,
                 b_init, c_init):
        self.in_size = in_size
        self.st_size = st_size
        self.out_size = out_size

        self.W = w_init.new_matrix((st_size, st_size))
        self.U = u_init.new_matrix((st_size, in_size))
        self.b = b_init.new_matrix(st_size)
        self.V = v_init.new_matrix((out_size, st_size))
        self.c = c_init.new_matrix(out_size)

        self.w_grad = np.empty_like(self.W)
        self.u_grad = np.empty_like(self.U)
        self.b_grad = np.empty_like(self.b)
        self.v_grad = np.empty_like(self.V)
        self.c_grad = np.empty_like(self.c)

        self.t_steps = 1
        self.seq_nce = np.empty((self.t_steps, in_size))
        self.prev_st = np.empty((self.t_steps + 1, st_size))
        self.probabs = np.empty((self.t_steps, out_size))

    def w_grad_pair(self):
        yield (self.W, self.w_grad, 'W')
        yield (self.U, self.u_grad, 'U')
        yield (self.b, self.b_grad, 'b')
        yield (self.V, self.v_grad, 'V')
        yield (self.c, self.c_grad, 'c')

    def fwd(self, seq_nce, previous_st):
        assert seq_nce.shape[1] == self.in_size
        assert previous_st.size == self.st_size
        self.t_steps = seq_nce.shape[0]

        self.seq_nce = seq_nce
        self.prev_st = np.empty((self.t_steps + 1, self.st_size))
        self.probabs = np.empty((self.t_steps, self.out_size))

        for t in range(self.t_steps):
            self.prev_st[t] = previous_st
            self.probabs[t], previous_st = self.pred_probabs(seq_nce[t],
                                                          previous_st)
        self.prev_st[-1] = previous_st

        return self.probabs, self.prev_st

    def bwd(self, targs):
        assert self.probabs.shape == targs.shape
        dL_do = self.probabs - targs

        self.c_grad = dL_do.sum(axis=0)

        self.v_grad = np.zeros_like(self.V)
        for t in range(self.t_steps):
            self.v_grad += np.outer(dL_do[t], self.prev_st[t+1])

        self.w_grad = np.zeros_like(self.W)
        self.u_grad = np.zeros_like(self.U)
        self.b_grad = np.zeros_like(self.b)
        dL_da = np.zeros(self.st_size)
        for t in range(self.t_steps - 1, 0 - 1, -1):
            dL_dh = dL_do[t] @ self.V + dL_da @ self.W
            dL_da = dL_dh * (1 - self.prev_st[t + 1] ** 2)

            self.w_grad += np.outer(dL_da, self.prev_st[t])
            self.u_grad += np.outer(dL_da, self.seq_nce[t])
            self.b_grad += dL_da

    def _softmax(self, o):
        try:
            e = np.exp(o)
            res = e / e.sum()
        except FloatingPointError:
            res = np.full_like(o, fill_value=np.finfo(float).eps)
            res[np.argmax(o)] = 1 - \
                                (self.out_size - 1) * np.finfo(float).eps
        return res
    
    def pred_probabs(self, x, previous_st):
        assert x.size == self.in_size
        assert previous_st.size == self.st_size
        a = self.W @ previous_st + self.U @ x + self.b
        h = np.tanh(a)
        o = self.V @ h + self.c
        p = self._softmax(o)
        return p, h

    def pred_class(self, x, previous_st):
        probabs, previous_st = self.pred_probabs(x, previous_st)
        one_h = np.zeros(self.out_size)
        one_h[np.random.choice(self.out_size, p=probabs)] = 1
        return one_h, previous_st
    
    def cost(self, targs):
        assert self.probabs.shape == targs.shape
        log_arg = (self.probabs * targs).sum(axis=1)
        log_arg[log_arg == 0] = np.finfo(float).eps
        return - np.log(log_arg).sum()

    def __str__(self):
        return 'RNN {} -> {} -> {}'.format(
            self.in_size, self.st_size, self.out_size)

class charRneuralnet(Rneuralnet):
    def __init__(self, inout_size, st_size,
                 w_init, u_init, v_init,
                 b_init, c_init):
        super().__init__(inout_size, inout_size, st_size,
                         w_init, u_init, v_init,
                         b_init, c_init)

    def gen(self, x, previous_st, t_steps):
        res = np.empty((t_steps, self.out_size), dtype=np.int)
        for t in range(t_steps):
            res[t], previous_st = self.pred_class(x, previous_st)
            x = res[t]
        return res, previous_st

def gen_text(rnn, data, firstc=None, init_st=None, length=200):
    if firstc is None:
        firstc = np.zeros(rnn.in_size)
        firstc[np.random.randint(0, rnn.in_size)] = 1
    if init_st is None:
        init_st = np.zeros(rnn.st_size)
    seq, last_st = rnn.gen(firstc, init_st, length)
    return data.dec(seq), last_st

def crnn_save(rnn, fname):
    np.savez_compressed(
        fname,
        **{nam: weight for weight, _, nam in rnn.w_grad_pair()},
        inout_size=rnn.in_size,
        st_size=rnn.st_size
    )

def crnn_restore(fname):
    got = np.load(fname)
    rnn = charRneuralnet(
        inout_size=got['inout_size'],
        st_size=got['st_size'],
        w_init=Zeros(),
        u_init=Zeros(),
        v_init=Zeros(),
        b_init=Zeros(),
        c_init=Zeros(),
    )
    for weight, _, nam in rnn.w_grad_pair():
        weight += got[nam]
    return rnn

    def get_seq(t_steps, in_size, out_size):
        np.random.seed(123)
        in_seq = np.zeros((t_steps, in_size))
        out_seq = np.zeros((t_steps, out_size))
        for t in range(t_steps):
            in_seq[t, np.random.randint(0, in_size)] = 1
            out_seq[t, np.random.randint(0, out_size)] = 1
        return in_seq, out_seq

    @staticmethod
    def compute_grads_for_matrix(ins, targs, init_st, matx,
                                 nw, nam):
        matx = np.atleast_2d(matx)
        grad = np.empty_like(matx)
        h = 1e-4

        for i in range(matx.shape[0]):
            for j in range(matx.shape[1]):
                matx[i, j] += h
                nw.fwd(ins, init_st)
                pos_c = nw.cost(targs)
                matx[i, j] -= 2 * h
                nw.fwd(ins, init_st)
                neg_c = nw.cost(targs)
                grad[i, j] = (pos_c - neg_c) / (2 * h)
                matx[i, j] += h
        return np.squeeze(grad)

    def grad_diff_out(self, grad, grad_num, nam=''):
        er = np.abs(grad - grad_num)
        reler = er / np.maximum(np.finfo('float').eps,
                                   np.abs(grad) + np.abs(grad_num))
        print('Grad values {}: {:.2e}'.format(nam, np.max(reler)))

        bad_coor = np.unravel_index(np.argmax(reler), reler.shape)
        self.assertLess(reler[bad_coor], 1e-4,
                        msg='grad {:.3e}\nnum {:.3e}\ndiff {:.3e}\n'
                        .format(grad[bad_coor], grad_num[bad_coor],
                                er[bad_coor]))

class Rneuralnetopt(ABC):
    def __init__(self, rnn, st_ful=False):
        self.rnn = rnn
        self.smooth_c = []
        self.st_ful = st_ful
        self.steps = 0

    @abstractmethod
    def train(self, seq_pairs, epochs=1):
        pass

    def metric_up(self, cost):
        self.steps += 1
        if len(self.smooth_c) > 0:
            cost = .999 * self.smooth_c[-1] + .001 * cost
        self.smooth_c.append(cost)

class vanilla(Rneuralnetopt):
    def __init__(self, rnn, eta_init,
                 decay_factor=1.0, st_ful=False, clip=None):
        super().__init__(rnn, st_ful)
        self.eta = eta_init
        self.decay_factor = decay_factor
        self.clip = clip
        self.eta_s = []

    def train(self, seq_pairs, epochs=1,
              call_b=None, callb_every=0,
              callb_epoch=None):
        for _ in range(epochs):
            previous_st = np.zeros(self.rnn.st_size)
            for i, sp in enumerate(seq_pairs):
                probabs, states = self.rnn.fwd(sp.ip, previous_st)
                self.rnn.bwd(sp.op)
                self.exec_up()
                cost = self.rnn.cost(sp.op)
                self.metric_up(cost)
                if self.st_ful:
                    previous_st = states[-1]
                else:
                    previous_st = np.zeros(self.rnn.st_size)
                if call_b is not None and i % callb_every == 0:
                    call_b(self, probabs[-1], states[-1])
                self.eta *= self.decay_factor
            if callb_epoch:
                callb_epoch(self)

    def exec_up(self):
        for weights, grad, nam in self.rnn.w_grad_pair():
            if self.clip is not None:
                grad = np.clip(grad, -self.clip, +self.clip)
            update = self.eta * grad
            weights -= update

    def metric_up(self, cost):
        super().metric_up(cost)
        self.eta_s.append(self.eta)

    @staticmethod
    def eta_warn(param, update):
        p_scale = np.linalg.norm(param.ravel())
        if p_scale != 0:
            up_scale = np.linalg.norm(update.ravel())
            ratio = up_scale / p_scale
            if ratio > 1e-2:
                print('Ratio update', ratio, 'eta too high')
            elif ratio < 1e-4:
                print('Ratio update', ratio, 'eta too low')

class Rneuralnet_rms(Rneuralnet_his):
    def __init__(self, rnn, eta_init,
                 decay_factor=1.0, st_ful=False, gamma=0.0, clip=None):
        super().__init__(rnn, eta_init,
                         decay_factor, st_ful, clip)
        self.gamma = gamma

    def exec_up(self):
        for weights, grad, nam in self.rnn.w_grad_pair():
            if self.clip is not None:
                grad = np.clip(grad, -self.clip, +self.clip)
            self.hist[nam] = self.gamma * self.hist[nam] + \
                              (1 - self.gamma) * grad ** 2
            update = self.eta * grad / \
                     np.sqrt(self.hist[nam] + np.finfo(float).eps)
            weights -= update
